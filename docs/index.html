button.py 
from pyblade import liveblade

class Button(liveblade.Component):
props = {
    'type': str,
    'size': str
}

def render(self):
    classes = self.get_classes()
    return f"""
        <button class="{classes}">
            {self.slots.icon}
            {self.slots.default}
        </button>
    """
